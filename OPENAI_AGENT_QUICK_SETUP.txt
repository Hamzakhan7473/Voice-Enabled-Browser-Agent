    post:
      operationId: executeBrowserTask
      summary: Execute browser automation task
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              required: [goal]
              properties:
                goal:
                  type: string
                  description: What to accomplish
                startUrl:
                  type: string
                  description: Optional starting URL
                maxSteps:
                  type: integer
                  default: 30
      responses:
        '200':
          description: Task result
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                  result:
                    type: string
                  steps:
                    type: integer
